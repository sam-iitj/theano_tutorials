class word2vec:
    
    def __init__(self, v, d):
        self.V = v
        self.D = d 
        
        We = init_weight(self.V, self.D)
        Wo = init_weight(self.D, self.V)
        
        self.We = theano.shared(We)
        self.Wo = theano.shared(Wo)
        
        self.params = [self.We, self.Wo]
        
    def fit(self, X, learning_rate=10e-5, mu=0.99, reg=1.0, activation=T.tanh, epochs=50, show_fig=True):
        a = T.iscalar('a')
        b = T.iscalar('b')
        
        h = self.We[a] 
        o = T.nnet.softmax(h.dot(self.Wo))
        
        prediction = T.argmax(o) 
        self.predict_op = theano.function(inputs=[a], 
                                          outputs=prediction)
        
        cost = -T.mean(T.log(o[:, b]))
        grads = T.grad(cost, self.params)
        dparams = [theano.shared(p.get_value()*0) for p in self.params]
        lr = T.scalar('learning_rate')
        
        updates = [
            (p, p + mu*dp - lr*g) for p, dp, g in zip(self.params, dparams, grads)
        ] + [
            (dp, mu*dp - lr*g) for dp, g in zip(dparams, grads)
        ]
        
        self.train_op = theano.function(
            inputs=[a, b, lr],
            outputs=[cost, prediction],
            updates=updates,
            allow_input_downcast=True,
        )
        
        costs = []
        for i in xrange(epochs):
            X = shuffle(X)
            n_correct = 0
            cost = 0
            for j in xrange(len(X)):
                c, p = self.train_op(X[j][0], X[j][1], learning_rate)
                cost += c
                if p == X[j][1]:
                    n_correct += 1

            learning_rate *= 0.9999
            
            print "i:", i, "cost:", cost, "correct rate:", (float(n_correct)/len(X)),
            print " "
            costs.append(cost)

        #if show_fig:
        #    plt.plot(costs)
        #    plt.show()

from util import get_wikipedia_data

def get_wiki_data():
    sentences, word2idx = get_wikipedia_data(n_files=10, n_vocab=500, by_paragraph=True)
    training_data = []
    vocab_size = len(word2idx)
    for sentence in sentences:
        for elem1, elem2 in zip(sentence[:-1], sentence[1:]):
            training_data.append((elem1, elem2))
    training_data = list(set(training_data))
    idx2word = {v:k for k, v in word2idx.iteritems()}
    return len(word2idx), training_data, word2idx, idx2word
